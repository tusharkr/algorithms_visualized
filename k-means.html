## Run K Means Algorithm ##
class KMeans():
    
    def __init__(self, X, n_means=5):
        self.X = X
        self.mu = [(np.random.uniform(-6,6),np.random.uniform(-6,6)) for i in xrange(n_means)]
        self.newmu = []
        #self.myplot()

    def myplot(self, colors_list):
        plt.gca().cla() 
        plt.scatter(self.X[:,0], self.X[:,1], c=colors_list)
        # b: blue g: green r: red c: cyan m: magenta y: yellow k: black w: white
        plt.scatter([c[0] for c in centers], [c[1] for c in centers], s=1000, c='y')
        plt.scatter([m[0] for m in self.mu], [m[1] for m in self.mu], s=1000, c='r')
        display.clear_output(wait=True)
        display.display(plt.gcf())
        time.sleep(1.0)
        
    def reevaluate_centers(self):
        labels = {}
        lengths = {}
        for i, x in enumerate(self.X):
            v  = labels.get(labels_true[i], np.zeros(len(x))) + x
            l  = lengths.get(labels_true[i], 0) + 1
            labels[labels_true[i]] = v
            lengths[labels_true[i]] = l
        #returning  new means
        self.mu = [(v[0]/lengths[k],v[1]/lengths[k]) for k,v in labels.items()]
        return

    def has_converged(self):
        return (set([tuple(a) for a in self.newmu]) == set([tuple(a) for a in self.mu]))

    
    def run(self):
        k = 0
        while k < 20:
            for i,x in enumerate(self.X):
                labels_true[i] = min([(np.linalg.norm(x-self.mu[j]),j) for j,_ in enumerate(self.mu)], key=lambda t:t[0])[1]
            self.reevaluate_centers()
            colors_list = [colors[label] for label in labels_true]
            k += 1
            self.myplot(colors_list)
